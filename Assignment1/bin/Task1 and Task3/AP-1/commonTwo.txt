public int commonTwo(String[] a, String[] b) {
  int i=0,j=0;
  int count=0;
  String last="";
  while(i<a.length && j<b.length){
    if(a[i].compareTo(b[j])<0){
      i++;
    }
    else if(a[i].compareTo(b[j])>0){
      j++;
    }
    else{
      if(!a[i].equals(last)){
        last=a[i];
        count++;
      }
      i++;
      j++;
    }
  }
  return count;
}
