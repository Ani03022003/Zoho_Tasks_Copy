public String[] wordsWithout(String[] words, String target) {
  int len=0;
  for(int i=0;i<words.length;i++){
    if(words[i]!=target)
      len++;
  }
  String arr[]=new String[len];
  int k=0;
  for(int i=0;i<words.length;i++){
    if(words[i]!=target){
      arr[k++]=words[i];
    }
  }
  return arr;
}
