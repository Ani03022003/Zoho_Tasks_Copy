public String[] fizzBuzz(int start, int end) {
  int len=end-start;
  int k=0;
  String arr[]=new String[len];
  for(int i=start;i<end;i++){
    if(i%3==0 && i%5==0){
      arr[k++]="FizzBuzz";
    }
    else if(i%3==0){
      arr[k++]="Fizz";
    }
    else if(i%5==0){
      arr[k++]="Buzz";
    }
    else{
      arr[k++]=String.valueOf(i);
    }
  }
  return arr;
}
