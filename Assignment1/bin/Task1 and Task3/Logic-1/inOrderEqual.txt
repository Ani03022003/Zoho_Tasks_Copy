public boolean inOrderEqual(int a, int b, int c, boolean equalOk) {
  if(equalOk){
    if(a==b && b==c)
      return true;
    else if(a==b && c>b)
      return true;
    else if(b==c && b>a)
      return true;
    else 
      return false;
  }
  else{
    if(b>a && c>b)
      return true;
    else
      return false;
  }
}
