public boolean evenlySpaced(int a, int b, int c) {
  if(a>=b && a>=c){
    if(c>=b){
      int temp=c;
      c=b;
      b=temp;
    }
  }
  else if(b>=a && b>=c){
    int temp=b;
    b=a;
    a=temp;
    if(c>=b){
      temp=c;
      c=b;
      b=temp;
    }
  }
  else if(c>=a && c>=b){
    int temp=c;
    c=a;
    a=temp;
    if(c>=b){
      temp=c;
      c=b;
      b=temp;
    }
  }
  if(Math.abs(a-b) == Math.abs(b-c))
    return true;
  return false;
}
