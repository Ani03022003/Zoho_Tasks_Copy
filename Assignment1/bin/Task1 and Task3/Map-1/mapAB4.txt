public Map<String, String> mapAB4(Map<String, String> map) {
  if(!map.containsKey("a") || !map.containsKey("b"))
    return map;
  int aLen=map.get("a").length();
  int bLen=map.get("b").length();
  if(aLen!=bLen){
    if(aLen>bLen){
      map.put("c",map.get("a"));
    }
    else{
      map.put("c",map.get("b"));
    }
  }
  else if(aLen==bLen){
    map.put("a","");
    map.put("b","");
  }
  return map;
}
