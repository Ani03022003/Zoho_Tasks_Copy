public String[] allSwap(String[] strings) {
  Map<Character, Integer> map = new HashMap<>();
  for(int i=0;i<strings.length;i++){
    String currentString = strings[i];
    char firstChar = currentString.charAt(0);
    if (map.containsKey(firstChar)) {
      int j = map.get(firstChar);
      String temp = strings[i];
      strings[i] = strings[j];
      strings[j] = temp;
      map.remove(firstChar);
    } 
    else{
      map.put(firstChar, i);
    }
  }
  return strings;
}

public String[] allSwap(String[] strings) {
  Map<String, Boolean> swapped=new HashMap<>();
  for(int i=0;i<strings.length;i++){
    swapped.put(strings[i],false);
  }
  for(int i=0;i<strings.length-1;i++){
    for(int j=i+1;j<strings.length;j++){
      if(strings[i].charAt(0)==strings[j].charAt(0) && !swapped.get(strings[i]) && !swapped.get(strings[j])){
        String temp=strings[j];
        strings[j]=strings[i];
        strings[i]=temp;
        swapped.put(strings[i],true);
        swapped.put(strings[j],true);
      }
    }
  }
  return strings;
}
