public int countX(String str) {
  if(str.length()==0)
    return 0;
  if(str.length()==1 && str.charAt(0)=='x')
    return 1;
  if(str.length()==1 && str.charAt(0)!='x')
    return 0;
  if(str.charAt(0)=='x')
    return 1+countX(str.substring(1,str.length()));
  return countX(str.substring(1,str.length()));
}
