public int maxBlock(String str) {
  if(str.length()==0)
    return 0;
  int max_count=1;
  for(int i=0;i<str.length();i++)
  {
    if(i<str.length()-1 && str.charAt(i)==str.charAt(i+1)){
      int j=i;
      int count=1;
      while(j<str.length()-1){
        if(str.charAt(j)==str.charAt(j+1)){
          count++;
          if(count>max_count)
            max_count=count;
        }
        else{
          break;
        }
        j++;
      }
      i+=count-1;
    }
  }
  return max_count;
}
