public String stringX(String str) {
  String result="";
  if(str.length()<=1)
    return str;
  for(int i=0;i<str.length();i++){
    if(str.charAt(i)!='x'){
      result+=str.charAt(i);
    }
  }
  if(str.charAt(0)=='x' && str.charAt(str.length()-1)=='x')
    return "x"+result+"x";
  return result;
}
