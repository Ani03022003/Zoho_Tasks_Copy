public String wordAppend(String[] strings) {
  Map<String, Integer> map = new HashMap<>();
  String result="";
  for(int i=0;i<strings.length;i++){
    map.put(strings[i],map.getOrDefault(strings[i],0)+1);
    int count=map.get(strings[i]);
    if(count%2==0)
    {
      result+=strings[i];
    }
  }
  return result;
}
