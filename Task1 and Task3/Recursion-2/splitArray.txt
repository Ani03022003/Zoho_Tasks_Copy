public boolean splitArray(int[] nums) {
  return splitSum(nums,0,0,0);
}
public boolean splitSum(int[] nums,int start,int sum1,int sum2){
  if(start>=nums.length){
    return sum1==sum2;
  }
  if(splitSum(nums,start+1,sum1+nums[start],sum2))
    return true;
  if(splitSum(nums,start+1,sum1,sum2+nums[start]))
    return true;
  return false;
}
