public boolean gHappy(String str) {
  boolean happy=true;
  for(int i=0;i<str.length();i++){
    if(i>0 && i<str.length()-1 && str.charAt(i)=='g'){
      if(str.charAt(i-1)!='g' && str.charAt(i+1)!='g')
        happy=false;
    }
    else if(i==0){
      if(str.length()==1 && str.charAt(0)=='g')
        happy=false;
    }
    else if(i==str.length()-1 && str.charAt(i)=='g'){
      if(str.charAt(i-1)!='g')
        happy=false;
    }
  }
  return happy;
}
